example 

class Test
{
	public static void main(String args[])
	{
		int a[]; 	// declartion

		a = new int[3]; 	// memory allocation

		a[0] = 1; 		// assign
		a[1] = 2;
		a[2] = 3;

		try
		{
			for (int i = 0; i < a.length; i++)
				System.out.print("\n index["+i+"] = "+a[i]);
		}
		catch(ArrayIndexOutOfBoundsException error)
		{
			System.out.print(error);
		}

	}
}
